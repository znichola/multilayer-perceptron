# 'input_shape' and 'output_shape' are derived from the data, and
# all mlp classes (Sequential, Dense, Normalize, Sigmoid, SoftMax, ...)
# It's easy the write an invalid config, so don't be dumb.

model = Sequential([
    Normalize(),
    Dense(input_shape, 24, activation=Sigmoid, weight_init=HeUniform),
    Dense(24, 24, activation=Sigmoid, weight_init=HeUniform),
    Dense(24, 24, activation=Sigmoid, weight_init=HeUniform),
    Dense(24, output_shape, activation=SoftMax, weight_init=HeUniform),
])

# CategoricalCrossentropy
model.compile(loss=CategoricalCrossentropy, lr=0.1, epochs=200, batch=50, patience=0)
